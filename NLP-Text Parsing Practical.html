import spacy

# Load the english pipeline
nlp = spacy.load('en_core_web_sm')

# Data 
data = """
Data analysis is the process of inspecting, cleansing, transforming, and modeling data with the goal of discovering useful information, informing 
conclusions, and supporting decision-making. Data analysis has multiple facets and approaches, encompassing diverse techniques under a
variety of names, and is used in different business, science, and social science domains.
In today's business world, data analysis plays a role in making decisions more scientific and helping businesses operate more effectively
"""

# To preprocess the data, pass it to nlp english pipeline
# Process text
doc = nlp(data)

for chunk in doc.noun_chunks:
  print("NP:", chunk.text, " | Head:", chunk.root.text, " | Label:", chunk.root.dep_)
